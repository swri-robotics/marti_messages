cmake_minimum_required(VERSION 2.8.3)
project(marti_nav_msgs)

set(MSG_DEPS
  geometry_msgs
  marti_common_msgs
  std_msgs
)

set(BUILD_DEPS
  ${MSG_DEPS}
  message_generation
)

set(RUNTIME_DEPS
  ${MSG_DEPS}
  message_runtime
)

find_package(catkin REQUIRED COMPONENTS ${BUILD_DEPS})

add_message_files(FILES
  Command.msg
  LeadVehicle.msg
  PathPlanning.msg
  Route.msg
  RoutePoint.msg
  RoutePosition.msg
  RouteSpeed.msg
  RouteSpeedArray.msg
  TeleopState.msg
  VehicleControl.msg
)

add_service_files(FILES
  GetRoute.srv
  GetRouteList.srv
  SaveRecordedRoute.srv
  SetRoute.srv
)

generate_messages(DEPENDENCIES ${MSG_DEPS})

catkin_package(CATKIN_DEPENDS ${RUNTIME_DEPS})
